153  // TODO(rsoussan): Replace PinholeCamera with PinholePose?
154  Vector evaluateError(const Pose3& pose, boost::optional<Matrix&> H1 = boost::none) const {
155  try {
156  if (body_P_sensor_) {
157  if (H1) {
158  Matrix H0;
159  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_, H0), *K_);
160  Point2 reprojectionError(camera.project(landmark_point_, H1, boost::none, boost::none) - measured_);
161  *H1 = *H1 * H0;
162 
163  Matrix11 d_silu_d_x, d_silu_d_y;
164  reprojectionError.x() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.x(), bbox_.x() / 2.0, d_silu_d_x);
165  reprojectionError.y() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.y(), bbox_.y() / 2.0, d_silu_d_y);
166  H1->block<1, 6>(0, 0) = d_silu_d_x * H1->block<1, 6>(0, 0);
167  H1->block<1, 6>(1, 0) = d_silu_d_y * H1->block<1, 6>(1, 0);
168  return reprojectionError;
169  } else {
170  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_), *K_);
171  Point2 reprojectionError(camera.project(landmark_point_, H1, boost::none, boost::none) - measured_);
172  reprojectionError.x() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.x(), bbox_.x() / 2.0);
173  reprojectionError.y() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.y(), bbox_.y() / 2.0);
174  return reprojectionError;
175  }
176  } else {
177  PinholeCamera<CALIBRATION> camera(pose, *K_);
178  Point2 reprojectionError(camera.project(landmark_point_, H1, boost::none, boost::none) - measured_);
179  reprojectionError.x() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.x(), bbox_.x() / 2.0);
180  reprojectionError.y() = graph_localizer::SiluWithOffsetTwoWay(reprojectionError.y(), bbox_.y() / 2.0);
181  return reprojectionError;
182  }
183  } catch (CheiralityException& e) {
184  if (H1) *H1 = Matrix::Zero(2, 6);
185  if (verboseCheirality_)
186  std::cout << e.what() << ": Landmark moved behind camera " << DefaultKeyFormatter(this->key()) << std::endl;
187  if (throwCheirality_) throw CheiralityException(this->key());
188  }
189  return Vector2::Constant(0.0);
190  }

192  bool cheiralityError(const Pose3& pose) const {
193  try {
194  if (body_P_sensor_) {
195  PinholeCamera<CALIBRATION> camera(pose.compose(*body_P_sensor_), *K_);
196  camera.project(landmark_point_) - measured_;
197  } else {
198  PinholeCamera<CALIBRATION> camera(pose, *K_);
199  camera.project(landmark_point_) - measured_;
200  }
201  } catch (CheiralityException& e) {
202  return true;
203  }
204  return false;
205  }
